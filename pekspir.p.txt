"PROGRAM PEKSPIR;

CONST PI=3.141592653;
VAR POSX,POSY,UHEL:REAL;

#I D8:ZELVXY.I

PROCEDURE SPIRALEK(KROK,SKOK,KAT:REAL);
BEGIN
 IF KROK>=SKOK THEN
 BEGIN
  FD(KROK);
  RT(KAT);
  SPIRALEK(KROK-SKOK,SKOK,KAT);
 END;
END;

PROCEDURE SPIRALA(KROK,SKOK,KAT:REAL); 
VAR PX,PY,KATPOC:REAL;
BEGIN
 PX:=POSX;PY:=POSY;KATPOC:=HEADING;
 SPIRALEK(KROK,SKOK,KAT);
 PU;
 SETPOS(PX,PY);
 PD;
 SETH(KATPOC);
END;

PROCEDURE PEKSPIRAL(ILE,KROK,SKOK,KAT:REAL);
VAR I:INTEGER;
BEGIN
 ILE:=ROUND(ILE);
 FOR I:=1 TO ILE DO
 BEGIN
  SPIRALA(KROK,SKOK,KAT);
  RT(360/ILE);
 END;
END;

BEGIN
 CS;PD;
 PEKSPIRAL(5,500,12.5,122);
 PU;
 HOME;
 CLOSEXY;
END.
"

PROGRAM VLOCKY;

CONST PI=3.14159;
VAR POSX,POSY,UHEL:REAL;

#I D8:ZELVXY.I

PROCEDURE BUNKA(SMER,N,A:REAL);
BEGIN
 IF N=0 THEN
 BEGIN
  SETH(SMER);
  FD(A);
 END
 ELSE
 BEGIN
  BUNKA(SMER,N-1,A/3);
  BUNKA(SMER+60,N-1,A/3);
  BUNKA(SMER-60,N-1,A/3);
  BUNKA(SMER,N-1,A/3);
 END;
END;

PROCEDURE VLOCKA(N,A:REAL);
BEGIN
 BUNKA(0,N,A);
 BUNKA(240,N,A);
 BUNKA(120,N,A);
END;

BEGIN
 CS;
 SETPOS(400,-800);
 PD;
 VLOCKA(4,1458);
 PU;
 CLOSEXY;
END.
